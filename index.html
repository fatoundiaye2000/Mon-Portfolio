// Génération automatique des PDF de candidature
function generateApplicationPDF($studentId, $universityId) {
    require_once('tcpdf/tcpdf.php');
    
    // Récupération des données
    $student = getStudentData($studentId);
    $university = getUniversityData($universityId);
    
    // Création du PDF
    $pdf = new TCPDF(PDF_PAGE_ORIENTATION, PDF_UNIT, PDF_PAGE_FORMAT, true, 'UTF-8', false);
    $pdf->AddPage();
    
    // Contenu HTML
    $html = '<h1>Candidature '.$university['name'].'</h1>';
    $html .= '<p>Étudiant: '.$student['name'].'</p>';
    // ... autres données
    
    $pdf->writeHTML($html, true, false, true, false, '');
    
    // Enregistrement
    $filename = 'candidatures/'.$studentId.'_'.$universityId.'.pdf';
    $pdf->Output($filename, 'F');
    
    return $filename;
}
// Système anti-doublon pour les emails
function sendApplicationEmail($applicationId, $recipient) {
    $lockFile = "locks/email_".$applicationId.".lock";
    
    if (file_exists($lockFile)) {
        logError("Tentative d'envoi en double bloquée: ".$applicationId);
        return false;
    }
    
    // Création du verrou
    file_put_contents($lockFile, date('Y-m-d H:i:s'));
    
    try {
        $mail = new PHPMailer(true);
        // Configuration de l'email...
        $mail->send();
        logSuccess("Email envoyé: ".$applicationId);
        return true;
    } finally {
        // Suppression du verrou
        unlink($lockFile);
    }
}

interface ProtectedRouteProps {

  requiredRole?: string;
}

export const ProtectedRoute: React.FC<ProtectedRouteProps> = ({
  children,
  requiredRole
}) => {
  const { isAuthenticated, isLoading, hasRole } = useAuth();

  if (isLoading) {
    return (
      <div className="flex items-center justify-center min-h-screen">
        <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-blue-600"></div>
      </div>
    );
  }

  if (!isAuthenticated) {
    return <Navigate to="/login" replace />;
  }

  if (requiredRole && !hasRole(requiredRole)) {
    return (
      <div className="flex items-center justify-center min-h-screen">
        <div className="text-center">
          <h1 className="text-2xl font-bold text-red-600 mb-4">
            Accès Refusé
          </h1>
          <p className="text-gray-600">
            Vous n'avez pas les permissions nécessaires.
          </p>
        </div>
      </div>
    );
  }

  return <>{children}</>;
}
import { useState, useEffect, useCallback } from 'react';
import authService from '../services/auth.service';
import { RegisterRequest } from '../types/auth.types';

interface UseAuthReturn {
  token: string | null;
  user: { username: string; roles: string[] } | null;
  isAuthenticated: boolean;
  isLoading: boolean;
  error: string | null;
  login: (username: string, password: string) => Promise<void>;
  register: (data: RegisterRequest) => Promise<void>;
  logout: () => void;
  clearError: () => void;
  hasRole: (role: string) => boolean;
}

export const useAuth = (): UseAuthReturn => {
  const [token, setToken] = useState<string | null>(null);
  const [user, setUser] = useState<{ username: string; roles: string[] } | null>(null);
  const [isLoading, setIsLoading] = useState<boolean>(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    const initAuth = () => {
      try {
        const storedToken = authService.getToken();
        const storedUser = authService.getUser();
        if (storedToken && authService.isAuthenticated()) {
          setToken(storedToken);
          setUser(storedUser);
        } else {
          authService.logout();
        }
      } catch (err) {
        console.error('Erreur initialisation auth:', err);
        authService.logout();
      } finally {
        setIsLoading(false);
      }
    };
    initAuth();
  }, []);

  const login = useCallback(async (username: string, password: string) => {
    setIsLoading(true);
    setError(null);
    try {
      const decoded = await authService.login(username, password);
      const newToken = authService.getToken();
      setToken(newToken);
      setUser({ username: decoded.sub, roles: decoded.roles });
    } catch (err: unknown) {
      const errorMessage = err instanceof Error ? err.message : 'Erreur de connexion';
      setError(errorMessage);
      throw err;
    } finally {
      setIsLoading(false);
    }
  }, []);

  const register = useCallback(async (data: RegisterRequest) => {
    setIsLoading(true);
    setError(null);
    try {
      await authService.register(data);
      await login(data.nom, data.motDePasse);
    } catch (err: unknown) {
      const errorMessage = err instanceof Error ? err.message : 'Erreur d\'inscription';
      setError(errorMessage);
      throw err;
    } finally {
      setIsLoading(false);
    }
  }, [login]);

  const logout = useCallback(() => {
    authService.logout();
    setToken(null);
    setUser(null);
    setError(null);
  }, []);

  const clearError = useCallback(() => { setError(null); }, []);
  const hasRole = useCallback((role: string): boolean => authService.hasRole(role), []);
  const isAuthenticated = !!token && !!user;

  return { token, user, isAuthenticated, isLoading, error, login, register, logout, clearError, hasRole };
}
import httpClient from '../utils/httpClient';
import { API_ENDPOINTS, STORAGE_KEYS } from '../config/constants';
import { LoginRequest, LoginResponse, RegisterRequest, DecodedToken } from '../types/auth.types';
import { Utilisateur } from '../types/user.types';

class AuthService {
  async login(username: string, password: string): Promise<DecodedToken> {
    try {
      const loginData: LoginRequest = { email: username, password: password };
      const response = await httpClient.post<LoginResponse>(API_ENDPOINTS.LOGIN, loginData);
      const { token } = response.data;
      this.setToken(token);
      const decoded = this.decodeToken(token);
      this.setUser(decoded);
      return decoded;
    } catch (error) {
      console.error('Erreur login:', error);
      if (error instanceof Error) throw new Error(error.message);
      else throw new Error('Erreur de connexion');
    }
  }

  async register(data: RegisterRequest): Promise<Utilisateur> {
    try {
      const response = await httpClient.post<Utilisateur>(API_ENDPOINTS.USERS, data);
      return response.data;
    } catch (error) {
      console.error('Erreur inscription:', error);
      if (error instanceof Error) throw new Error(error.message);
      else throw new Error("Erreur lors de l'inscription");
    }
  }

  logout(): void {
    localStorage.removeItem(STORAGE_KEYS.TOKEN);
    localStorage.removeItem(STORAGE_KEYS.USER);
  }

  decodeToken(token: string): DecodedToken {
    try {
      const payloadBase64 = token.split('.')[1];
      const payloadJson = atob(payloadBase64);
      const decoded: DecodedToken = JSON.parse(payloadJson);
      return decoded;
    } catch (error) {
      console.error('Erreur décodage token:', error);
      throw new Error('Token invalide');
    }
  }

  isTokenExpired(token: string): boolean {
    try {
      const decoded = this.decodeToken(token);
      const now = Date.now() / 1000;
      return decoded.exp < now;
    } catch { return true; }
  }

  getToken(): string | null { return localStorage.getItem(STORAGE_KEYS.TOKEN); }
  private setToken(token: string): void { localStorage.setItem(STORAGE_KEYS.TOKEN, token); }
  private setUser(decoded: DecodedToken): void {
    const user = { username: decoded.sub, roles: decoded.roles };
    localStorage.setItem(STORAGE_KEYS.USER, JSON.stringify(user));
  }
  getUser(): { username: string; roles: string[] } | null {
    const userStr = localStorage.getItem(STORAGE_KEYS.USER);
    if (!userStr) return null;
    try { return JSON.parse(userStr); } catch { return null; }
  }
  isAuthenticated(): boolean {
    const token = this.getToken();
    if (!token) return false;
    return !this.isTokenExpired(token);
  }
  hasRole(role: string): boolean {
    const user = this.getUser();
    if (!user) return false;
    return user.roles.includes(role);
  }
}

export default new AuthService();

const handleLogin = async () => {
    // Étape 1 : POST /login → récupère { token, roles }
    const response = await fetch(`${BASE_URL}/login`, {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
            'Accept': 'application/json',
        },
        body: JSON.stringify({ email: emailTrimmed, password }),
    });

    const data = await response.json();
    const token = data.token;

    // Étape 2 : Récupérer idUser depuis /api/users/all
    const usersRes = await fetch(`${BASE_URL}/api/users/all`, {
        headers: { 'Authorization': `Bearer ${token}` },
    });
    const users = await usersRes.json();
    const found = users.find(u => u.email === emailTrimmed);

    // Étape 3 : Sauvegarder dans AsyncStorage
    await AsyncStorage.setItem('token',     token);
    await AsyncStorage.setItem('userId',    String(found.idUser));
    await AsyncStorage.setItem('userRoles', JSON.stringify(data.roles));

    // Naviguer vers la liste des événements
    navigation.replace('EventsList');
};
// EventDetailScreen.js — Bouton Participer
const handleParticiper = async () => {
    const token  = await AsyncStorage.getItem('token');
    const userId = await AsyncStorage.getItem('userId');

    if (!token || !userId) {
        Alert.alert('Non connecté', 'Veuillez vous reconnecter.');
        navigation.replace('Login');
        return;
    }

    const today = new Date().toISOString().split('T')[0];

    const res = await fetch(`${BASE_URL}/api/reservations/save`, {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
            'Authorization': `Bearer ${token}`,
        },
        body: JSON.stringify({
            dateReservation: today,
            nbrPersonnes:    1,
            user:      { idUser: parseInt(userId) },
            evenement: { idEvent: eventId },
        }),
    });

    // Gestion des codes HTTP
    if (res.status === 200 || res.status === 201) {
        Alert.alert('🎉 Inscription confirmée !');
        navigation.goBack();
    } else if (res.status === 401) {
        await AsyncStorage.multiRemove(['token', 'userId']);
        navigation.replace('Login');
    } else if (res.status === 403) {
        Alert.alert('Accès refusé', 'Rôle ADMIN requis.');
    } else {
        Alert.alert('Erreur réseau', 'Serveur inaccessible.');
    }
};

import * as SQLite from 'expo-sqlite';

const db = SQLite.openDatabaseSync('evenix_favoris.db');

// Initialiser la table au démarrage
export const initFavorisDB = () => {
    db.execSync(`
        CREATE TABLE IF NOT EXISTS favoris (
            id         INTEGER PRIMARY KEY AUTOINCREMENT,
            idEvent    INTEGER NOT NULL UNIQUE,
            titreEvent TEXT,
            typeEvent  TEXT,
            dateDebut  TEXT,
            villeEvent TEXT,
            addedAt    TEXT DEFAULT (datetime('now'))
        );
    `);
};

// Basculer favori (toggle ❤️ / 🤍)
export const toggleFavori = (event) => {
    if (isFavori(event.idEvent)) {
        db.runSync(
            'DELETE FROM favoris WHERE idEvent = ?',
            [event.idEvent]
        );
        return false;
    } else {
        db.runSync(
            `INSERT OR IGNORE INTO favoris
             (idEvent, titreEvent, typeEvent, dateDebut, villeEvent)
             VALUES (?, ?, ?, ?, ?)`,
            [event.idEvent, event.titreEvent,
             event.typeEvent?.nomType || '',
             event.dateDebut, event.villeEvent || '']
        );
        return true;
    }
};

// Vérifier si un événement est favori
export const isFavori = (idEvent) => {
    const result = db.getFirstSync(
        'SELECT id FROM favoris WHERE idEvent = ?', [idEvent]
    );
    return result !== null;
};

// Récupérer tous les favoris
export const getAllFavoris = () =>
    db.getAllSync(
        'SELECT * FROM favoris ORDER BY addedAt DESC'
    );

I. Évolution des Pare-feux : Du Filtrage Simple aux NGFW

Définition : Qu'est-ce qu'un NGFW ?

Un pare-feu de nouvelle génération (NGFW) est un dispositif de sécurité qui traite et filtre le trafic réseau avec des capacités avancées dépassant les pare-feux traditionnels. Les NGFW possèdent des fonctionnalités supplémentaires qui permettent d'identifier les dernières menaces de manière plus efficace.

Source : Cloudflare Learning Center

1990s

Firewalls Stateless (1ère génération)

Fonctionnement : Filtrage basique basé uniquement sur les adresses IP et les numéros de ports

Limitation : Analyse limitée aux couches 3-4 du modèle OSI (réseau et transport)

Efficacité : Protection minimale contre les menaces modernes

2000s

Firewalls Stateful (2ème génération)

Fonctionnement : Suivi de l'état des connexions réseau avec mémorisation du contexte

Innovation : Inspection plus approfondie des paquets avec analyse contextuelle

Avantage : Meilleure détection des connexions illégitimes

2010s

NGFW - Next Generation Firewall (3ème génération)

Fonctionnement : Inspection profonde jusqu'à la couche 7 (applicative)

Capacités : Identification des applications, intégration IPS/IDS, contrôle des utilisateurs

Innovation majeure : Définition formalisée par Gartner en 2012

2024-2025

NGFW + Intelligence Artificielle (4ème génération)

Fonctionnement : Détection comportementale pilotée par l'IA et Machine Learning

Capacités : Analyse prédictive, réponse automatisée, détection des menaces zero-day

Source : Nomios, décembre 2024 & Cloudflare

II. Fonctionnalités Clés des NGFW Modernes

🔍

Inspection Profonde des Paquets (DPI)

Analyse complète du contenu des paquets au-delà des en-têtes. Les NGFW examinent les données applicatives et peuvent détecter des menaces cachées dans le contenu.

Source : Cloudflare Learning Center
🎯

Contrôle des Applications

Identification et contrôle des applications indépendamment des ports et protocoles utilisés. Permet de bloquer des applications spécifiques.

Source : Cloudflare & Insyncom
🛡️

Système de Prévention d'Intrusion (IPS)

Détection et blocage automatique des menaces connues et potentielles via signatures et analyse comportementale.

Source : Cloudflare & Insyncom
🧠

Intelligence des Menaces

Intégration de flux de renseignements sur les menaces en temps réel. Maintient à jour les signatures de malwares.

Source : Cloudflare
👤

Identification Utilisateur

Contrôle d'accès basé sur l'identité des utilisateurs et non uniquement sur les adresses IP.

Source : Insyncom
🔐

Déchiffrement SSL/TLS

Inspection du trafic chiffré pour détecter les menaces cachées dans les communications HTTPS.

Source : Nomios, décembre 2024

IV. Zero Trust Architecture

Qu'est-ce que le Zero Trust ?

Le Zero Trust est une approche de sécurité basée sur le principe "Ne jamais faire confiance, toujours vérifier". Contrairement aux modèles traditionnels qui font confiance à tout ce qui est à l'intérieur du périmètre réseau, le Zero Trust assume que des menaces peuvent exister partout.
